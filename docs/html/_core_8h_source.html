562 struct Command {
563 enum class Types {
564 None = 0,
565 Point,
566 Line,
567 Rectangle,
568 Ellipse,
569 Fill = 8,
570 Spirit = 16
571 } graphics;
572 explicit Command() : graphics(Types::None) {};
573 virtual ~Command() = default;
574 virtual void exec(SRenderer *renderer, uint32_t thickness) = 0;
575 };
576 struct PointCMD : Command {
577 Graphics::Point pt;
578 explicit PointCMD(const Graphics::Point& point) : pt(point) { Command::graphics = Command::Types::Point; }
579 void exec(SRenderer *renderer, uint32_t thickness) override;
580 };
581 struct LineCMD : Command {
582 Graphics::Line line;
583 explicit LineCMD(const Graphics::Line& line) : line(line) { Command::graphics = Command::Types::Line; }
584 void exec(SRenderer *renderer, uint32_t thickness) override;
585 };
586 struct RectCMD : Command {
587 Graphics::Rectangle rect;
588 explicit RectCMD(const Graphics::Rectangle& rect) : rect(rect) { Command::graphics = Command::Types::Rectangle; }
589 void exec(SRenderer *renderer, uint32_t thickness) override;
590 };
591 struct EllipseCMD : Command {
592 Graphics::Ellipse ellipse;
593 explicit EllipseCMD(const Graphics::Ellipse& ellipse) : ellipse(ellipse)
594 { Command::graphics = Command::Types::Ellipse; }
595 void exec(SRenderer *renderer, uint32_t thickness) override;
596 };
597 struct FillCMD : Command {
598 SColor color;
599 explicit FillCMD(const SColor& color) : color(color)
600 { Command::graphics = Command::Types::Fill; }
601 void exec(SRenderer *renderer, uint32_t thickness) override;
602 };
603 struct SpriteCMD : Command {
604 STexture* _sprite;

614 SDL_FlipMode _flip_mode{SDL_FLIP_NONE};
615 explicit SpriteCMD(const Components::Sprite& sprite, const Vector2& pos = Vector2(0, 0))
616 : _sprite(sprite.spirit()), _size(sprite.size()), _pos(pos),
617 _rotate_center(0, 0), _scaled_center(0, 0),
618 _clip_pos(0, 0), _clip_size(0, 0)
619 { Command::graphics = Command::Types::Spirit; }

629 }
