647 };

652 };

662 };

667 void exec(SRenderer *renderer, uint32_t thickness) override;
668 };

676 STexture* _sprite;

689 _rotate_center(0, 0), _scaled_center(0, 0),
690 _clip_pos(0, 0), _clip_size(0, 0)

701 }
